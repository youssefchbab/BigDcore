    bge x5, x0, positive    # 0002D463  | if x5 >= 0, branch
    addi x8, x0, -1         # FFF00413  | shouldn't execute
    jal x0, 8               # 0080006F  | skip
    
positive:
    addi x8, x0, 1          # 00100413  | x8 = 1 (positive marker)

    sra x19, x3, x16        # 4101D9B3  | x19 = -5 >> 2 (arith) = -2
    slli x20, x5, 3         # 00329A13  | x20 = 30 << 3 = 240

    # SECTION 5: Comparison and Branching

    slti x25, x1, 15        # 00F0AC93  | x25 = (10 < 15) = 1
    sltiu x26, x3, 10       # 00A1BD13  | x26 = (-5 < 10 unsigned) = 0

    beq x1, x1, equal_test  # 00108463  | branch (should take)
    addi x27, x0, 99        # 06300D93  | shouldn't execute
    
equal_test:
    addi x27, x0, 1         # 00100D93  | x27 = 1 (branch taken)
    
    bne x1, x2, not_equal   # 00209463  | branch (should take)
    addi x28, x0, 99        # 06300E13  | shouldn't execute
    
not_equal:
    addi x28, x0, 2         # 00200E13  | x28 = 2 (branch taken)

    lw x29, 0(x10)          # 00052E83  | x29 = mem[x10+0] (should be 30)
    lw x30, 4(x10)          # 00452F03  | x30 = mem[x10+4] (should be 10)

    # SECTION 7: Function Call Simulation (JAL/JALR)

    addi x2, x0, 100        # 06400113  | x2 = 100 (argument)

    jal x1, function        # 0280006F  | call function, save return
    
    # Return point - verify result
    addi x3, x0, 0xAA       # 0AA00193  | x3 = 0xAA (after return marker)
    jal x0, end_test        # 0200006F  | jump to end
    
function:
    # Function: double the input (x2) and return
    add x2, x2, x2          # 00210133  | x2 = x2 * 2 = 200
    addi x4, x0, 0xBB       # 0BB00213  | x4 = 0xBB (function marker)
    
    # Return using JALR
    jalr x0, x1, 0          # 00008067  | return to caller
    
    # ============================================
    # SECTION 8: Loop with Branch
    # ============================================
end_test:
    addi x5, x0, 0          # 00000293  | x5 = 0 (counter)
    addi x6, x0, 5          # 00500313  | x6 = 5 (limit)
    
loop_start:
    addi x5, x5, 1          # 00128293  | x5++
    blt x5, x6, loop_start  # FE62C4E3  | if x5 < 5, continue loop
    
    # After loop, x5 should be 5
    addi x7, x0, 0xCC       # 0CC00393  | x7 = 0xCC (loop done marker)
    
    # ============================================
    # SECTION 9: Complex Computation
    # ============================================
    # Compute: result = (a + b) * 4 - c
    # where a=x1(100 from load), b=10, c=5
    addi x8, x0, 10         # 00A00413  | x8 = 10
    addi x9, x0, 5          # 00500493  | x9 = 5
    
    add x11, x1, x8         # 00808533  | x11 = x1 + 10
    slli x11, x11, 2        # 00259593  | x11 = x11 * 4
    sub x11, x11, x9        # 409585B3  | x11 = x11 - 5
    
    # ============================================
    # SECTION 10: Conditional Store
    # ============================================
    # If x11 > 100, store it; otherwise store 0
    addi x12, x0, 100       # 06400613  | x12 = 100
    blt x12, x11, store_val # 00B64463  | if 100 < x11, branch
    
    # Store 0
    sw x0, 12(x10)          # 00052623  | mem[x10+12] = 0

store_val:
    sw x11, 12(x10)         # 00B52623  | mem[x10+12] = x11

    # SECTION 11: Final verification markers

    addi x31, x0, 0xFF      # 0FF00F93  | x31 = 255 (SUCCESS)
